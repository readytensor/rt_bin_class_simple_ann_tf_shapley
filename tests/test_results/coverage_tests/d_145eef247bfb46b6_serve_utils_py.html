84async def transform_req_data_and_make_predictions( 

85 data: pd.DataFrame, model_resources: ModelResources, request_id: str 

86) -> Tuple[pd.DataFrame, dict]: 

87 """Transform request data and generate predictions based on request. 

88 

89 Function performs the following steps: 

90 1. Validate the request data 

91 2. Transforms the dataframe using preprocessing pipeline 

92 3. Makes predictions as np array on the transformed data using the predictor model 

93 4. Converts predictions np array into pandas dataframe with required structure 

94 5. Converts the predictions dataframe into a dictionary with required structure 

95 

96 Args: 

97 request (InferenceRequestBodyModel): The request body containing the input data. 

98 model_resources (ModelResources): Resources needed by inference service. 

99 request_id (str): Unique request id for logging and tracking 

100 

101 Returns: 

102 Tuple[pd.DataFrame, dict]: Tuple containing transformed data and 

103 prediction response. 

104 """ 

105 logger.info(f"Predictions requested for {len(data)} samples...") 

106 

107 # validate the data 

108 logger.info("Validating data...") 

109 validated_data = validate_data( 

110 data=data, data_schema=model_resources.data_schema, is_train=False 

111 ) 

112 

113 logger.info("Transforming data sample(s)...") 

114 transformed_data, _ = transform_data( 

115 model_resources.preprocessor, model_resources.target_encoder, validated_data 

116 ) 

117 

118 logger.info("Making predictions...") 

119 predictions_arr = predict_with_model( 

120 model_resources.predictor_model, transformed_data, return_probs=True 

121 ) 

122 logger.info("Converting predictions array into dataframe...") 

123 predictions_df = create_predictions_dataframe( 

124 predictions_arr, 

125 model_resources.data_schema.target_classes, 

126 model_resources.model_config["prediction_field_name"], 

127 data[model_resources.data_schema.id], 

128 model_resources.data_schema.id, 

129 return_probs=True, 

130 ) 

131 

132 logger.info("Converting predictions dataframe into response dictionary...") 

133 predictions_response = create_predictions_response( 

134 predictions_df, model_resources.data_schema, request_id 

135 ) 

136 return transformed_data, predictions_response

139def create_predictions_response( 

140 predictions_df: pd.DataFrame, data_schema: Any, request_id: str 

141) -> Dict[str, Any]: 

142 """ 

143 Convert the predictions DataFrame to a response dictionary in required format. 

144 

145 Args: 

146 transformed_data (pd.DataFrame): The transfomed input data for prediction. 

147 data_schema (Any): An instance of the BinaryClassificationSchema. 

148 request_id (str): Unique request id for logging and tracking 

149 

150 Returns: 

151 dict: The response data in a dictionary. 

152 """ 

153 class_names = data_schema.target_classes 

154 # find predicted class which has the highest probability 

155 predictions_df["__predicted_class"] = predictions_df[class_names].idxmax(axis=1) 

156 sample_predictions = [] 

157 for sample in predictions_df.to_dict(orient="records"): 

158 sample_predictions.append( 

159 { 

160 "sampleId": sample[data_schema.id], 

161 "predictedClass": str(sample["__predicted_class"]), 

162 "predictedProbabilities": [ 

163 round(sample[class_names[0]], 5), 

164 round(sample[class_names[1]], 5), 

165 ], 

166 } 

167 ) 

168 predictions_response = { 

169 "status": "success", 

170 "message": "", 

171 "timestamp": pd.Timestamp.now().isoformat(), 

172 "requestId": request_id, 

173 "targetClasses": class_names, 

174 "targetDescription": data_schema.target_description, 

175 "predictions": sample_predictions, 

176 } 

177 return predictions_response

180def combine_predictions_response_with_explanations( 

181 predictions_response: dict, explanations: dict 

182) -> dict: 

183 """ 

184 Combine the predictions response with explanations. 

185 

186 Inserts explanations for each sample into the respective prediction dictionary 

187 for the sample. 

188 

189 Args: 

190 predictions_response (dict): The response data in a dictionary. 

191 explanations (dict): The explanations for the predictions. 

192 """ 

193 for pred, exp in zip( 

194 predictions_response["predictions"], explanations["explanations"] 

195 ): 

196 pred["explanation"] = exp 

197 predictions_response["explanationMethod"] = explanations["explanation_method"] 

198 return predictions_response
